Domanda 1: Si consideri il seguente codice JavaScript

JAVASCRIPT
<script type="text/javascript">
  var metadati = [{"isbn":"isbn1", "prezzo": "20euro"},
                  {"isbn":"isbn2", "prezzo": "30euro"},
                  {"isbn":"isbn3", "prezzo": "15euro"},
                  {"isbn":"isbn4", "prezzo": "50euro"}];

  var libro = [{"isbn":"isbn1", "edizione": 10, "titolo": "Anna Karenina", "autore": "Leo Tolstoy"},
               {"isbn":"isbn2", "edizione": 3, "titolo": "Madame Bovary", "autore": "Gustave Flaubert"},
               {"isbn":"isbn3", "edizione": 7, "titolo": "Hamlet", "autore": "William Shakespeare"},
               {"isbn":"isbn4", "edizione": 5, "titolo": "Divina Commedia", "autore": "Dante Alighieri"}];

  localStorage.setItem("md", JSON.stringify (metadati));
  localStorage.setItem("book", JSON.stringify(libro));
  
  var chiave = localStorage.key(0);
  var tmp = JSON.parse(localStorage.getItem(chiave));
  
  for (i=0; i<metadati.length;i++)
    if (metadati[i].isbn==tmp[2].isbn)
      alert(tmp[2].prezzo);
</script>
  • 1. alert stampa 20euro
  • 2. alert stampa 30euro
  • 3. alert stampa 15euro
  • 4. alert stampa undefined
